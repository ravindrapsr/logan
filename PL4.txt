DECLARE
	SAL STAFF_MASTER.STAFF_SAL%TYPE;
	DEPT STAFF_MASTER.DEPT_CODE%TYPE;
	CURSOR CUR IS SELECT MAX(STAFF_SAL) FROM STAFF_MASTER WHERE DEPT_CODE=DEPT;
BEGIN
	DEPT:=&DEPT;
	OPEN CUR;
	LOOP
		FETCH CUR INTO SAL;
		EXIT WHEN CUR %NOTFOUND;
		DBMS_OUTPUT.PUT_LINE(SAL);
	END LOOP;
	CLOSE CUR;
END;
/

2.
CREATE OR REPLACE FUNCTION FIND_AGE(D DATE)
	RETURN NUMBER IS 
	AGE NUMBER;
BEGIN
	AGE:=FLOOR(MONTHS_BETWEEN(SYSDATE,D)/12);
	RETURN AGE;
END FIND_AGE;
/


DECLARE
	DOB DATE;
BEGIN
	DOB:='&DOB';
	DBMS_OUTPUT.PUT_LINE(FIND_AGE(DOB));
END;
/

3.
CREATE OR REPLACE PROCEDURE UPP_NAME(SC NUMBER)
	IS
	CODE STAFF_MASTER.STAFF_CODE %TYPE;
	CURSOR CUR3 IS SELECT STAFF_CODE FROM STAFF_MASTER WHERE STAFF_CODE=SC; 
BEGIN
	OPEN CUR3;
	LOOP
		FETCH CUR3 INTO CODE;
		EXIT WHEN CUR3%NOTFOUND;
		UPDATE STAFF_MASTER
		SET STAFF_NAME=UPPER(STAFF_NAME)
		WHERE STAFF_CODE=CODE;

	END LOOP;
END UPP_NAME;
/

DECLARE
	CODE NUMBER;
BEGIN
	CODE:=&CODE;
	UPP_NAME(CODE);
END;
/
